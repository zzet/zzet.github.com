# –Ч–∞–Ї–Њ–Љ–Љ–Є—В–Є—В—М –≤—Б–µ –Є–Ј–Љ–µ–љ–µ–љ–Є—П –≤ —Г–ґ–µ –Њ—В—Б–ї–µ–ґ–Є–≤–∞–µ–Љ—Л—Е —Д–∞–є–ї–∞—Е

–Т–∞–ґ–љ–Њ–µ –Њ–≥—А–∞–љ–Є—З–µ–љ–Є–µ: -a –љ–µ –і–Њ–±–∞–≤–ї—П–µ—В –љ–Њ–≤—Л–µ —Д–∞–є–ї—Л (untracked). –Ґ–Њ–ї—М–Ї–Њ –Є–Ј–Љ–µ–љ–µ–љ–Є—П –Є —Г–і–∞–ї–µ–љ–Є—П –≤ —Г–ґ–µ –Њ—В—Б–ї–µ–ґ–Є–≤–∞–µ–Љ—Л—Е —Д–∞–є–ї–∞—Е.

git commit -am "Add new feature"

# –Ф–Њ–±–∞–≤–Є—В—М –≤—Б—С (–≤–Ї–ї—О—З–∞—П –љ–Њ–≤—Л–µ —Д–∞–є–ї—Л) –Є –Ј–∞–Ї–Њ–Љ–Љ–Є—В–Є—В—М

# –І—В–Њ –і–µ–ї–∞–µ—В -A:
# вЬУ –Ш–Ј–Љ–µ–љ—С–љ–љ—Л–µ —Д–∞–є–ї—Л
# вЬУ –£–і–∞–ї—С–љ–љ—Л–µ —Д–∞–є–ї—Л
# вЬУ –Э–Њ–≤—Л–µ (untracked) —Д–∞–є–ї—Л
# вЬУ –Я–µ—А–µ–Є–Љ–µ–љ–Њ–≤–∞–љ–љ—Л–µ —Д–∞–є–ї—Л

–°–њ–Њ—Б–Њ–± 3: git add . && git commit #

# –Ф–Њ–±–∞–≤–Є—В—М –Є–Ј —В–µ–Ї—Г—Й–µ–є –њ–∞–њ–Ї–Є –Є –љ–Є–ґ–µ

–Т —Б–Њ–≤—А–µ–Љ–µ–љ–љ—Л—Е –≤–µ—А—Б–Є—П—Е Git (2.x) git add . –≤–µ–і—С—В —Б–µ–±—П –∞–љ–∞–ї–Њ–≥–Є—З–љ–Њ git add -A вАФ –і–Њ–±–∞–≤–ї—П–µ—В –≤—Б—С, –≤–Ї–ї—О—З–∞—П —Г–і–∞–ї—С–љ–љ—Л–µ —Д–∞–є–ї—Л. –Т —Б—В–∞—А—Л—Е –≤–µ—А—Б–Є—П—Е Git git add . –љ–µ –і–Њ–±–∞–≤–ї—П–ї–Њ —Г–і–∞–ї—С–љ–љ—Л–µ —Д–∞–є–ї—Л.

–°—А–∞–≤–љ–µ–љ–Є–µ —Б–њ–Њ—Б–Њ–±–Њ–≤ #

git commit -a         вЬУ            вЬУ           вЬЧ

–Я—А–Њ–≤–µ—А–Ї–∞ –њ–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ #

–Я–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ –≤—Б–µ—Е –Є–Ј–Љ–µ–љ–µ–љ–Є–є вАФ –њ–Њ—Б–Љ–Њ—В—А–µ—В—М —З—В–Њ –Є–Љ–µ–љ–љ–Њ –њ–Њ–њ–∞–і—С—В:

# –Я–Њ—Б–Љ–Њ—В—А–µ—В—М —З—В–Њ –±—Г–і–µ—В –і–Њ–±–∞–≤–ї–µ–љ–Њ —З–µ—А–µ–Ј -a
git diff  # –≤—Б–µ –љ–µindex–Є—А–Њ–≤–∞–љ–љ—Л–µ –Є–Ј–Љ–µ–љ–µ–љ–Є—П

# –Я–Њ—Б–Љ–Њ—В—А–µ—В—М —З—В–Њ –њ–Њ–њ–∞–і—С—В –≤ –Ї–Њ–Љ–Љ–Є—В (–µ—Б–ї–Є —Г–ґ–µ git add)
git status
git diff --staged

# "–°—Г—Е–Њ–є –њ—А–Њ–≥–Њ–љ" вАФ –њ–Њ–Ї–∞–Ј—Л–≤–∞–µ—В —Д–∞–є–ї—Л –±–µ–Ј —А–µ–∞–ї—М–љ–Њ–≥–Њ add
git add --dry-run -A

–°–Њ—Е—А–∞–љ–µ–љ–Є–µ —З–∞—Б—В–Є –Є–Ј–Љ–µ–љ–µ–љ–Є–є (–Є–љ—В–µ—А–∞–Ї—В–Є–≤–љ–Њ–µ –і–Њ–±–∞–≤–ї–µ–љ–Є–µ) #

–Х—Б–ї–Є –љ—Г–ґ–љ–Њ –Ј–∞–Ї–Њ–Љ–Љ–Є—В–Є—В—М –љ–µ –≤—Б–µ, –∞ —В–Њ–ї—М–Ї–Њ —З–∞—Б—В—М –Є–Ј–Љ–µ–љ–µ–љ–Є–є:

# –Ш–љ—В–µ—А–∞–Ї—В–Є–≤–љ—Л–є –≤—Л–±–Њ—А
git add -p

# Git –±—Г–і–µ—В —Б–њ—А–∞—И–Є–≤–∞—В—М –њ–Њ –Ї–∞–ґ–і–Њ–Љ—Г –±–ї–Њ–Ї—Г –Є–Ј–Љ–µ–љ–µ–љ–Є–є:
# Stage this hunk? [y,n,q,a,d,/,e,?]
# y - –і–Њ–±–∞–≤–Є—В—М
# n - –њ—А–Њ–њ—Г—Б—В–Є—В—М
# q - –≤—Л–є—В–Є
# a - –і–Њ–±–∞–≤–Є—В—М –≤–µ—Б—М —Д–∞–є–ї
# d - –њ—А–Њ–њ—Г—Б—В–Є—В—М –≤–µ—Б—М —Д–∞–є–ї

–Р–ї–Є–∞—Б—Л –і–ї—П –±—Л—Б—В—А–Њ–≥–Њ –Ї–Њ–Љ–Љ–Є—В–∞ #

# –°–Њ–Ј–і–∞—В—М –∞–ї–Є–∞—Б –і–ї—П –±—Л—Б—В—А–Њ–≥–Њ –Ї–Њ–Љ–Љ–Є—В–∞ –≤—Б–µ–≥–Њ
git config --global alias.ca 'commit -am'

# –Ш—Б–њ–Њ–ї—М–Ј–Њ–≤–∞–љ–Є–µ
git ca "Quick update"

git config --global alias.aac '!git add -A && git commit -m'

git aac "Add all and commit"

–Я—А–∞–≤–Є–ї—М–љ—Л–є —А–∞–±–Њ—З–Є–є –њ—А–Њ—Ж–µ—Б—Б #

–•–Њ—В—П git commit -am —Г–і–Њ–±–љ–Њ, –Њ—Б–Њ–Ј–љ–∞–љ–љ—Л–є –њ—А–Њ—Ж–µ—Б—Б –љ–∞–і—С–ґ–љ–µ–µ:

# 1. –Я–Њ—Б–Љ–Њ—В—А–µ—В—М —З—В–Њ –Є–Ј–Љ–µ–љ–Є–ї–Њ—Б—М
git status
git diff

# 2. –Ф–Њ–±–∞–≤–Є—В—М —Д–∞–є–ї—Л –Њ—Б–Њ–Ј–љ–∞–љ–љ–Њ
git add src/feature.js
git add tests/feature.test.js
# (–љ–µ —В—А–Њ–≥–∞–µ–Љ debug.log –Є –≤—А–µ–Љ–µ–љ–љ—Л–µ —Д–∞–є–ї—Л)

# 3. –Я—А–Њ–≤–µ—А–Є—В—М —З—В–Њ –њ–Њ–њ–∞–і—С—В –≤ –Ї–Њ–Љ–Љ–Є—В
git status
git diff --staged

# 4. –°–Њ–Ј–і–∞—В—М –Ї–Њ–Љ–Љ–Є—В

–Р–≤—В–Њ–Љ–∞—В–Є—З–µ—Б–Ї–Є–є git add -A –Є–ї–Є git commit -a –Љ–Њ–ґ–µ—В —Б–ї—Г—З–∞–є–љ–Њ –≤–Ї–ї—О—З–Є—В—М –≤ –Ї–Њ–Љ–Љ–Є—В –ї–Є—И–љ–Є–µ —Д–∞–є–ї—Л (–≤—А–µ–Љ–µ–љ–љ—Л–µ, –Ї–Њ–љ—Д–Є–і–µ–љ—Ж–Є–∞–ї—М–љ—Л–µ, –Њ—В–ї–∞–і–Њ—З–љ—Л–µ). –£–±–µ–і–Є—В–µ—Б—М, —З—В–Њ .gitignore –љ–∞—Б—В—А–Њ–µ–љ –Ї–Њ—А—А–µ–Ї—В–љ–Њ.

–І–∞—Б—В–Њ –Ј–∞–і–∞–≤–∞–µ–Љ—Л–µ –≤–Њ–њ—А–Њ—Б—Л #

–Я–Њ—З–µ–Љ—Г git commit -a –љ–µ –і–Њ–±–∞–≤–ї—П–µ—В –љ–Њ–≤—Л–µ —Д–∞–є–ї—Л? –≠—В–Њ —Б–Њ–Ј–љ–∞—В–µ–ї—М–љ–Њ–µ —А–µ—И–µ–љ–Є–µ Git. –Э–Њ–≤—Л–µ —Д–∞–є–ї—Л —В—А–µ–±—Г—О—В —П–≤–љ–Њ–≥–Њ –і–Њ–±–∞–≤–ї–µ–љ–Є—П (git add), —З—В–Њ–±—Л —Б–ї—Г—З–∞–є–љ–Њ –љ–µ –Ј–∞–Ї–Њ–Љ–Љ–Є—В–Є—В—М –љ–µ–ґ–µ–ї–∞—В–µ–ї—М–љ–Њ–µ. –Я–Њ—Б–ї–µ –њ–µ—А–≤–Њ–≥–Њ git add —Д–∞–є–ї —Б—В–∞–љ–Њ–≤–Є—В—Б—П ¬Ђ–Њ—В—Б–ї–µ–ґ–Є–≤–∞–µ–Љ—Л–Љ¬ї –Є –±—Г–і–µ—В –њ–Њ–і—Е–≤–∞—В—Л–≤–∞—В—М—Б—П -a.

–Т —З—С–Љ —А–∞–Ј–љ–Є—Ж–∞ –Љ–µ–ґ–і—Г git add . –Є git add -A? –Т Git 2.x –Њ–љ–Є —Н–Ї–≤–Є–≤–∞–ї–µ–љ—В–љ—Л. –Т —Б—В–∞—А—Л—Е –≤–µ—А—Б–Є—П—Е (1.x) git add . –љ–µ –і–Њ–±–∞–≤–ї—П–ї–Њ —Г–і–∞–ї—С–љ–љ—Л–µ —Д–∞–є–ї—Л вАФ –і–ї—П –љ–Є—Е –љ—Г–ґ–µ–љ –±—Л–ї git add -u. –Т —Б–Њ–≤—А–µ–Љ–µ–љ–љ–Њ–Љ Git –Є—Б–њ–Њ–ї—М–Ј—Г–є—В–µ –ї—О–±–Њ–є –Є–Ј –≤–∞—А–Є–∞–љ—В–Њ–≤.

–Ъ–∞–Ї –њ—А–Њ–≤–µ—А–Є—В—М —З—В–Њ git commit -am –љ–µ –і–Њ–±–∞–≤–Є—В –ї–Є—И–љ–µ–≥–Њ? –Ч–∞–њ—Г—Б—В–Є—В–µ git status –њ–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ. –Я–Њ—Б–Љ–Њ—В—А–Є—В–µ —А–∞–Ј–і–µ–ї “Changes not staged” вАФ —Н—В–Њ —В–Њ, —З—В–Њ -a –і–Њ–±–∞–≤–Є—В. –†–∞–Ј–і–µ–ї “Untracked files” вАФ —Н—В–Њ —В–Њ, —З—В–Њ -a –Э–Х –і–Њ–±–∞–≤–Є—В.

–Ь–Њ–ґ–љ–Њ –ї–Є —Б–і–µ–ї–∞—В—М git add -A –±–µ–Ј–Њ–њ–∞—Б–љ—Л–Љ? –Ф–∞, —З–µ—А–µ–Ј —Е–Њ—А–Њ—И–Њ –љ–∞—Б—В—А–Њ–µ–љ–љ—Л–є .gitignore. –Х—Б–ї–Є –≤—Б–µ –љ–µ–ґ–µ–ї–∞—В–µ–ї—М–љ—Л–µ —Д–∞–є–ї—Л –≤ .gitignore вАФ git add -A –љ–µ –Ј–∞—Е–≤–∞—В–Є—В –Є—Е.

–Ш–љ—В–µ—А–∞–Ї—В–Є–≤–љ–Њ–µ –і–Њ–±–∞–≤–ї–µ–љ–Є–µ —Д–∞–є–ї–Њ–≤ (git add -p) #

–°–∞–Љ—Л–є –±–µ–Ј–Њ–њ–∞—Б–љ—Л–є —Б–њ–Њ—Б–Њ–± –і–Њ–±–∞–≤–Є—В—М –Є–Љ–µ–љ–љ–Њ —В–Њ —З—В–Њ –љ—Г–ґ–љ–Њ:

# –Ш–љ—В–µ—А–∞–Ї—В–Є–≤–љ—Л–є —А–µ–ґ–Є–Љ –≤—Л–±–Њ—А–∞ –Є–Ј–Љ–µ–љ–µ–љ–Є–є
git add -p
# –Є–ї–Є
git add --patch

# Git –±—Г–і–µ—В –њ–Њ–Ї–∞–Ј—Л–≤–∞—В—М –њ–Њ –Њ–і–љ–Њ–Љ—Г –±–ї–Њ–Ї—Г (hunk) –Є —Б–њ—А–∞—И–Є–≤–∞—В—М:
# Stage this hunk? [y,n,q,a,d,/,e,?]

# y - –і–Њ–±–∞–≤–Є—В—М –±–ї–Њ–Ї
# n - –њ—А–Њ–њ—Г—Б—В–Є—В—М
# q - –≤—Л–є—В–Є –Є–Ј —А–µ–ґ–Є–Љ–∞
# a - –і–Њ–±–∞–≤–Є—В—М –≤—Б–µ –±–ї–Њ–Ї–Є –≤ —Н—В–Њ–Љ —Д–∞–є–ї–µ
# d - –њ—А–Њ–њ—Г—Б—В–Є—В—М –≤—Б–µ –±–ї–Њ–Ї–Є –≤ —Н—В–Њ–Љ —Д–∞–є–ї–µ
# / - –њ–Њ–Є—Б–Ї –±–ї–Њ–Ї–∞
# e - —А–µ–і–∞–Ї—В–Є—А–Њ–≤–∞—В—М –±–ї–Њ–Ї –≤—А—Г—З–љ—Г—О
# ? - –њ–Њ–Љ–Њ—Й—М

–≠—В–Њ –Є–і–µ–∞–ї—М–љ—Л–є —Б–њ–Њ—Б–Њ–± –і–ї—П —А–∞–Ј—А–∞–±–Њ—В—З–Є–Ї–∞ –Ї–Њ—В–Њ—А—Л–є —Е–Њ—З–µ—В –њ–Њ–ї–љ–Њ—Б—В—М—О –Ї–Њ–љ—В—А–Њ–ї–Є—А–Њ–≤–∞—В—М —З—В–Њ –њ–Њ–њ–∞–і—С—В –≤ –Ї–Њ–Љ–Љ–Є—В:

# –Я—А–Є–Љ–µ—А —А–∞–±–Њ—З–µ–≥–Њ –њ—А–Њ—Ж–µ—Б—Б–∞
git add -p
# –Т—Л–±—А–∞—В—М —В–Њ–ї—М–Ї–Њ —А–µ–ї–µ–≤–∞–љ—В–љ—Л–µ –Є–Ј–Љ–µ–љ–µ–љ–Є—П
git status  # –њ—А–Њ–≤–µ—А–Є—В—М —З—В–Њ –і–Њ–±–∞–≤–Є–ї–Њ—Б—М
git diff --staged  # –њ–Њ—Б–Љ–Њ—В—А–µ—В—М —Д–Є–љ–∞–ї—М–љ–Њ–µ —Б–Њ–і–µ—А–ґ–Є–Љ–Њ–µ
git commit -m "feat: specific feature"

–†–∞–Ј–ї–Є—З–Є–µ –Љ–µ–ґ–і—Г git add -u, git add . –Є git add -A #

–Я–Њ–ї–љ–∞—П —В–∞–±–ї–Є—Ж–∞ –њ–Њ–≤–µ–і–µ–љ–Є—П:

# вЬУ –Ш–Ј–Љ–µ–љ—С–љ–љ—Л–µ —Д–∞–є–ї—Л

# вЬУ –Э–Њ–≤—Л–µ —Д–∞–є–ї—Л (Git 2.x+)

# –Р–ї–Є–∞—Б –і–ї—П –њ—А–Њ–≤–µ—А–Ї–Є –њ–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ

# –Я—А–Њ–≤–µ—А–Є—В –Є –њ–Њ–Ї–∞–ґ–µ—В —З—В–Њ –±—Г–і–µ—В –Ј–∞–Ї–Њ–Љ–Љ–Є—З–µ–љ–Њ

# –Я–Њ—Б–Љ–Њ—В—А–µ—В—М —З—В–Њ –≤ staging area

–І–∞—Б—В–Њ –Ј–∞–і–∞–≤–∞–µ–Љ—Л–µ –≤–Њ–њ—А–Њ—Б—Л #

–Я–Њ—З–µ–Љ—Г git commit -a –љ–µ –і–Њ–±–∞–≤–ї—П–µ—В –љ–Њ–≤—Л–µ —Д–∞–є–ї—Л? –≠—В–Њ —Б–Њ–Ј–љ–∞—В–µ–ї—М–љ–Њ–µ —А–µ—И–µ–љ–Є–µ Git. –Э–Њ–≤—Л–µ —Д–∞–є–ї—Л —В—А–µ–±—Г—О—В —П–≤–љ–Њ–≥–Њ –і–Њ–±–∞–≤–ї–µ–љ–Є—П (git add), —З—В–Њ–±—Л —Б–ї—Г—З–∞–є–љ–Њ –љ–µ –Ј–∞–Ї–Њ–Љ–Љ–Є—В–Є—В—М –љ–µ–ґ–µ–ї–∞—В–µ–ї—М–љ–Њ–µ. –Я–Њ—Б–ї–µ –њ–µ—А–≤–Њ–≥–Њ git add —Д–∞–є–ї —Б—В–∞–љ–Њ–≤–Є—В—Б—П ¬Ђ–Њ—В—Б–ї–µ–ґ–Є–≤–∞–µ–Љ—Л–Љ¬ї –Є –±—Г–і–µ—В –њ–Њ–і—Е–≤–∞—В—Л–≤–∞—В—М—Б—П -a.

–Т —З—С–Љ —А–∞–Ј–љ–Є—Ж–∞ –Љ–µ–ґ–і—Г git add . –Є git add -A? –Т Git 2.x –Њ–љ–Є —Н–Ї–≤–Є–≤–∞–ї–µ–љ—В–љ—Л. –Т —Б—В–∞—А—Л—Е –≤–µ—А—Б–Є—П—Е (1.x) git add . –љ–µ –і–Њ–±–∞–≤–ї—П–ї–Њ —Г–і–∞–ї—С–љ–љ—Л–µ —Д–∞–є–ї—Л вАФ –і–ї—П –љ–Є—Е –љ—Г–ґ–µ–љ –±—Л–ї git add -u. –Т —Б–Њ–≤—А–µ–Љ–µ–љ–љ–Њ–Љ Git –Є—Б–њ–Њ–ї—М–Ј—Г–є—В–µ –ї—О–±–Њ–є –Є–Ј –≤–∞—А–Є–∞–љ—В–Њ–≤.

–Ъ–∞–Ї –њ—А–Њ–≤–µ—А–Є—В—М —З—В–Њ git commit -am –љ–µ –і–Њ–±–∞–≤–Є—В –ї–Є—И–љ–µ–≥–Њ? –Ч–∞–њ—Г—Б—В–Є—В–µ git status –њ–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ. –Я–Њ—Б–Љ–Њ—В—А–Є—В–µ —А–∞–Ј–і–µ–ї “Changes not staged” вАФ —Н—В–Њ —В–Њ, —З—В–Њ -a –і–Њ–±–∞–≤–Є—В. –†–∞–Ј–і–µ–ї “Untracked files” вАФ —Н—В–Њ —В–Њ, —З—В–Њ -a –Э–Х –і–Њ–±–∞–≤–Є—В.

–Ь–Њ–ґ–љ–Њ –ї–Є —Б–і–µ–ї–∞—В—М git add -A –±–µ–Ј–Њ–њ–∞—Б–љ—Л–Љ? –Ф–∞, —З–µ—А–µ–Ј —Е–Њ—А–Њ—И–Њ –љ–∞—Б—В—А–Њ–µ–љ–љ—Л–є .gitignore. –Х—Б–ї–Є –≤—Б–µ –љ–µ–ґ–µ–ї–∞—В–µ–ї—М–љ—Л–µ —Д–∞–є–ї—Л –≤ .gitignore вАФ git add -A –љ–µ –Ј–∞—Е–≤–∞—В–Є—В –Є—Е.

–Ъ–∞–Ї –Њ—В–Љ–µ–љ–Є—В—М git add –њ–µ—А–µ–і –Ї–Њ–Љ–Љ–Є—В–Њ–Љ? git restore --staged filename —Г–±–Є—А–∞–µ—В –Є–Ј staging. –Ш–ї–Є git reset —Г–±–Є—А–∞–µ—В –≤—Б—С –Є–Ј staging.

–І–µ–Љ git add -p –ї—Г—З—И–µ git add -A? git add -p –њ–Њ–Ї–∞–Ј—Л–≤–∞–µ—В –Ї–∞–ґ–і–Њ–µ –Є–Ј–Љ–µ–љ–µ–љ–Є–µ –Є –њ—А–Њ—Б–Є—В –њ–Њ–і—В–≤–µ—А–ґ–і–µ–љ–Є–µ. –≠—В–Њ –≥–Њ—А–∞–Ј–і–Њ –±–µ–Ј–Њ–њ–∞—Б–љ–µ–µ –і–ї—П –Є–Ј–±–µ–ґ–∞–љ–Є—П —Б–ї—Г—З–∞–є–љ—Л—Е –і–Њ–±–∞–≤–ї–µ–љ–Є–є.

–І—В–Њ –µ—Б–ї–Є —Д–∞–є–ї –≤ .gitignore –љ–Њ —Е–Њ—З—Г –µ–≥–Њ –і–Њ–±–∞–≤–Є—В—М? git add -f filename вАФ –њ—А–Є–љ—Г–і–Є—В–µ–ї—М–љ–Њ –і–Њ–±–∞–≤–Є—В—М –і–∞–ґ–µ –Є–≥–љ–Њ—А–Є—А—Г–µ–Љ—Л–є —Д–∞–є–ї.

–Ч–∞–Ї–ї—О—З–µ–љ–Є–µ #

–Ф–ї—П –Ї–Њ–Љ–Љ–Є—В–∞ –≤—Б–µ—Е –Є–Ј–Љ–µ–љ–µ–љ–Є–є –≤ –Њ—В—Б–ї–µ–ґ–Є–≤–∞–µ–Љ—Л—Е —Д–∞–є–ї–∞—Е вАФ git commit -am "message". –Ф–ї—П –і–Њ–±–∞–≤–ї–µ–љ–Є—П –љ–Њ–≤—Л—Е —Д–∞–є–ї–Њ–≤ –≤–Љ–µ—Б—В–µ —Б–Њ –≤—Б–µ–Љ–Є –Є–Ј–Љ–µ–љ–µ–љ–Є—П–Љ–Є вАФ git add -A && git commit -m "message". –Ф–ї—П –Љ–∞–Ї—Б–Є–Љ–∞–ї—М–љ–Њ–є –±–µ–Ј–Њ–њ–∞—Б–љ–Њ—Б—В–Є –Є—Б–њ–Њ–ї—М–Ј—Г–є—В–µ git add -p –і–ї—П –Є–љ—В–µ—А–∞–Ї—В–Є–≤–љ–Њ–≥–Њ –≤—Л–±–Њ—А–∞. –Я–µ—А–µ–і –ї—О–±—Л–Љ –Љ–∞—Б—Б–Њ–≤—Л–Љ –і–Њ–±–∞–≤–ї–µ–љ–Є–µ–Љ вАФ –Ј–∞–њ—Г—Б–Ї–∞–є—В–µ git status –і–ї—П –њ—А–Њ–≤–µ—А–Ї–Є. –Я–Њ–Љ–љ–Є—В–µ —З—В–Њ .gitignore –њ—А–µ–і–Њ—В–≤—А–∞—Й–∞–µ—В —Б–ї—Г—З–∞–є–љ–Њ–µ –і–Њ–±–∞–≤–ї–µ–љ–Є–µ –љ–µ–ґ–µ–ї–∞—В–µ–ї—М–љ—Л—Е —Д–∞–є–ї–Њ–≤.
